```javascript
import AWS from 'aws-sdk'

let s3Client = null

export function getArticleS3Client() {
  if (!s3Client) {
    s3Client = new AWS.S3({
      endpoint: process.env.AWS_S3_ENDPOINT,
      s3ForcePathStyle: true,
      region: process.env.AWS_REGION,
    })
  }
  return s3Client
}

const getBucket = () => process.env.S3_ARTICLES_BUCKET_NAME

export async function putArticleJSON(key, articleData) { ... }
export async function getArticleJSON(key) { ... }

export async function putArticleIndex(indexData) { ... }  // writes articles/index.json
```

Key patterns:
- Same S3 client singleton with endpoint, forcePathStyle, region
- Same bucket env var pattern (S3_ARTICLES_BUCKET_NAME)
- getObject/putObject with .promise()
- JSON.stringify with null, 2 for pretty-print
- NoSuchKey handling returns empty index
- Missing bucket returns empty index (dev/build without S3)
